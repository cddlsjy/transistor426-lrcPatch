```kotlin
/* 设置遥控器方向键监听 */
private fun setupRemoteControl() {
    layout.recyclerView.isFocusable = true
    layout.recyclerView.isFocusableInTouchMode = true
    layout.recyclerView.setOnKeyListener { _, keyCode, event ->
        if (event.action == KeyEvent.ACTION_DOWN) {
            when (keyCode) {
                KeyEvent.KEYCODE_DPAD_UP -> {
                    moveSelection(-1)
                    return@setOnKeyListener true
                }
                KeyEvent.KEYCODE_DPAD_DOWN -> {
                    moveSelection(1)
                    return@setOnKeyListener true
                }
                KeyEvent.KEYCODE_DPAD_CENTER, KeyEvent.KEYCODE_ENTER -> {
                    onConfirmPressed()
                    return@setOnKeyListener true
                }
            }
        }
        false
    }
}

/* 移动选中项 */
private fun moveSelection(delta: Int) {
    val list = collection.stations
    if (list.isEmpty()) return

    var currentPos = collectionAdapter.getSelectedPosition()
    if (currentPos == -1) {
        currentPos = playerState.stationPosition
        if (currentPos !in 0 until list.size) currentPos = 0
    }

    var newPos = currentPos + delta
    if (newPos < 0) newPos = list.size - 1
    if (newPos >= list.size) newPos = 0

    collectionAdapter.setSelectedPosition(newPos)
    val station = list[newPos]

    // 滚动并聚焦
    scrollToPositionAndFocus(newPos)

    // 自动播放（如果开关开启）
    if (remoteAutoPlayEnabled) {
        onPlayButtonTapped(newPos)
    }
}

/* 确认键按下：播放/暂停 */
private fun onConfirmPressed() {
    val pos = collectionAdapter.getSelectedPosition()
    if (pos == -1) return

    val currentPlayingPos = playerState.stationPosition
    val isPlaying = controller?.isPlaying == true

    if (currentPlayingPos == pos && isPlaying) {
        // 当前选中的电台正在播放 → 暂停
        onPlayButtonTapped(pos)
    } else {
        // 否则播放
        onPlayButtonTapped(pos)
    }
}

/* 滚动到指定位置并请求焦点 */
private fun scrollToPositionAndFocus(position: Int) {
    layout.recyclerView.post {
        val targetView = layout.layoutManager.findViewByPosition(position)
        if (targetView != null) {
            targetView.requestFocus()
        } else {
            layout.recyclerView.smoothScrollToPosition(position)
            layout.recyclerView.postDelayed({
                layout.layoutManager.findViewByPosition(position)?.requestFocus()
            }, 150)
        }
    }
}

/* 供 CollectionAdapter 更新播放高亮（当播放状态改变时调用） */
fun setPlayingPosition(position: Int) {
    collectionAdapter.setPlayingPosition(position)
}
```
